Recreate Experiment 2 data loading and preprocessing from 0325.ipynb, then compare:

  • Cubic: trust ~ poly(lines, 3, raw = TRUE)

Metrics:

  • MSE
  • LOOCV-MSE
  • BIC
  • AIC
  • Lasso-regularized MSE

report_dir <- local({
  current_input <- tryCatch(
    knitr::current_input(),
    error = function(e) character(0)
  )

  if (length(current_input) == 1 && nzchar(current_input)) {
    input_path <- current_input

    if (!grepl("^(/|[A-Za-z]:[/\\\\])", input_path)) {
      input_path <- file.path(getwd(), input_path)
    }

    if (grepl("\\.[Rr]md$", input_path) || file.exists(input_path)) {
      return(normalizePath(dirname(input_path), winslash = "/", mustWork = TRUE))
    }
  }

  fallback_dir <- file.path(getwd(), "Experiment2", "analysis")
  normalizePath(fallback_dir, winslash = "/", mustWork = TRUE)
})

resolve_report_path <- function(path_value) {
  if (grepl("^(/|[A-Za-z]:[/\\\\])", path_value)) {
    return(normalizePath(path_value, winslash = "/", mustWork = TRUE))
  }

  normalizePath(
    file.path(report_dir, path_value),
    winslash = "/",
    mustWork = TRUE
  )
}

extract_condition_number <- function(condition_id) {
  if (is.na(condition_id) || !nzchar(as.character(condition_id))) {
    return(NA_real_)
  }

  raw <- sub(".*condition_([0-9]+).*", "\\1", as.character(condition_id), perl = TRUE)
  if (!grepl("^[0-9]+$", raw)) {
    return(NA_real_)
  }

  as.numeric(raw)
}

map_condition_labels <- function(condition_numbers) {
  vapply(
    condition_numbers,
    function(x) {
      key <- as.character(suppressWarnings(as.integer(x)))
      if (is.na(x) || !(key %in% names(CONDITION_LABELS))) {
        return("Other")
      }
      CONDITION_LABELS[[key]]
    },
    character(1)
  )
}

extract_participant_id_from_filename <- function(file_path) {
  parts <- strsplit(tools::file_path_sans_ext(basename(file_path)), "_", fixed = TRUE)[[1]]
  if (length(parts) >= 2) {
    return(parts[[2]])
  }
  NA_character_
}

parse_response_json <- function(value) {
  if (is.null(value) || length(value) == 0 || is.na(value) || !nzchar(as.character(value))) {
    return(list())
  }

  if (is.list(value) && !is.null(names(value))) {
    return(value)
  }

  parsed <- tryCatch(
    jsonlite::fromJSON(as.character(value), simplifyVector = TRUE),
    error = function(e) NULL
  )

  if (is.null(parsed) || !is.list(parsed)) {
    return(list())
  }

  parsed
}

safe_get_scalar <- function(x) {
  if (is.null(x) || length(x) == 0) {
    return(NA_character_)
  }

  as.character(x[[1]])
}

expand_response_columns <- function(data, response_col = "response") {
  if (!(response_col %in% names(data))) {
    return(data)
  }

  parsed_list <- lapply(data[[response_col]], parse_response_json)
  all_keys <- unique(unlist(lapply(parsed_list, names), use.names = FALSE))

  if (length(all_keys) == 0) {
    return(data)
  }

  expanded <- data.frame(
    lapply(all_keys, function(key) {
      vapply(parsed_list, function(item) safe_get_scalar(item[[key]]), character(1))
    }),
    stringsAsFactors = FALSE,
    check.names = FALSE
  )
  names(expanded) <- all_keys

  merged <- data
  for (col in names(expanded)) {
    parsed_values <- expanded[[col]]

    if (col %in% names(merged)) {
      existing <- merged[[col]]
      existing_chr <- trimws(as.character(existing))
      fill_mask <- is.na(existing) | existing_chr == ""
      existing[fill_mask] <- parsed_values[fill_mask]
      merged[[col]] <- existing
    } else {
      merged[[col]] <- parsed_values
    }
  }

  merged
}

identify_attention_check_failures <- function(
  data,
  trust_keys = TRUST_QUESTION_KEYS,
  interaction_keys = INTERACTION_QUESTION_KEYS,
  survey_trial_type = "trust-survey"
) {
  empty_result <- list(ids = character(0), details = data.frame())

  if (is.null(data) || nrow(data) == 0) {
    return(empty_result)
  }

  if (!all(c("trial_type", "participant_id") %in% names(data))) {
    return(empty_result)
  }

  survey_rows <- data[data$trial_type == survey_trial_type, , drop = FALSE]
  if (nrow(survey_rows) == 0) {
    return(empty_result)
  }

  survey_rows <- expand_response_columns(survey_rows)
  parsed_response <- if ("response" %in% names(survey_rows)) {
    lapply(survey_rows$response, parse_response_json)
  } else {
    replicate(nrow(survey_rows), list(), simplify = FALSE)
  }

  for (key in c(trust_keys, interaction_keys)) {
    existing <- if (key %in% names(survey_rows)) {
      suppressWarnings(as.numeric(survey_rows[[key]]))
    } else {
      rep(NA_real_, nrow(survey_rows))
    }

    if (sum(!is.na(existing)) == 0) {
      extracted <- vapply(parsed_response, function(item) safe_get_scalar(item[[key]]), character(1))
      existing <- suppressWarnings(as.numeric(extracted))
    }

    survey_rows[[key]] <- existing
  }

  trust_complete <- rowSums(is.na(survey_rows[, trust_keys, drop = FALSE])) == 0
  interaction_complete <- rowSums(is.na(survey_rows[, interaction_keys, drop = FALSE])) == 0

  trust_uniform <- apply(
    survey_rows[, trust_keys, drop = FALSE],
    1,
    function(row) {
      values <- unique(row[!is.na(row)])
      length(values) == 1
    }
  )

  interaction_uniform <- apply(
    survey_rows[, interaction_keys, drop = FALSE],
    1,
    function(row) {
      values <- unique(row[!is.na(row)])
      length(values) == 1
    }
  )

  failure_mask <- trust_complete & interaction_complete & trust_uniform & interaction_uniform
  failure_rows <- survey_rows[failure_mask & !is.na(survey_rows$participant_id), , drop = FALSE]

  if (nrow(failure_rows) == 0) {
    return(empty_result)
  }

  failure_rows$participant_id <- as.character(failure_rows$participant_id)
  detail_columns <- c(
    "participant_id",
    "condition_id",
    "condition_number",
    trust_keys,
    interaction_keys
  )
  available_detail_columns <- detail_columns[detail_columns %in% names(failure_rows)]

  list(
    ids = unique(failure_rows$participant_id),
    details = failure_rows[, available_detail_columns, drop = FALSE]
  )
}

normalize_scores_cousineau_morey <- function(
  data,
  metric_columns,
  baseline_condition = BASELINE_CONDITION_NUMBER,
  drop_baseline = TRUE,
  normalization_conditions = NULL,
  require_complete_participants = TRUE
) {
  if (is.null(data) || nrow(data) == 0) {
    normalized <- if (is.null(data)) data.frame() else data
    if (nrow(normalized) > 0) {
      normalized[[NORMALIZED_FLAG_COLUMN]] <- FALSE
      normalized[[MOREY_FACTOR_COLUMN]] <- NA_real_
      normalized[[K_CONDITIONS_COLUMN]] <- NA_real_
    }
    return(normalized)
  }

  normalized <- data

  if (!all(c("participant_id", "condition_number") %in% names(normalized))) {
    normalized[[NORMALIZED_FLAG_COLUMN]] <- FALSE
    normalized[[MOREY_FACTOR_COLUMN]] <- NA_real_
    normalized[[K_CONDITIONS_COLUMN]] <- NA_real_
    return(normalized)
  }

  available_columns <- metric_columns[metric_columns %in% names(normalized)]
  normalized$condition_number <- suppressWarnings(as.numeric(normalized$condition_number))
  normalized$.row_id <- seq_len(nrow(normalized))
  normalized[[NORMALIZED_FLAG_COLUMN]] <- FALSE
  normalized[[MOREY_FACTOR_COLUMN]] <- NA_real_
  normalized[[K_CONDITIONS_COLUMN]] <- NA_real_

  return_without_norm <- function(df) {
    df$.row_id <- NULL
    if (drop_baseline) {
      df <- df[df$condition_number != baseline_condition, , drop = FALSE]
    }
    df
  }

  if (length(available_columns) == 0) {
    return(return_without_norm(normalized))
  }

  if (is.null(normalization_conditions)) {
    observed_conditions <- sort(unique(stats::na.omit(as.integer(normalized$condition_number))))
    normalization_conditions <- observed_conditions[observed_conditions != as.integer(baseline_condition)]
  } else {
    normalization_conditions <- sort(unique(as.integer(normalization_conditions)))
  }

  if (length(normalization_conditions) == 0) {
    return(return_without_norm(normalized))
  }

  normalization_data <- normalized[normalized$condition_number %in% normalization_conditions, , drop = FALSE]
  if (nrow(normalization_data) == 0) {
    return(return_without_norm(normalized))
  }

  expected_k <- length(normalization_conditions)

  if (require_complete_participants) {
    participant_condition_counts <- stats::aggregate(
      condition_number ~ participant_id,
      data = normalization_data,
      FUN = function(x) length(unique(x[!is.na(x)]))
    )

    complete_participants <- as.character(
      participant_condition_counts$participant_id[
        participant_condition_counts$condition_number >= expected_k
      ]
    )

    normalization_data <- normalization_data[
      as.character(normalization_data$participant_id) %in% complete_participants,
      ,
      drop = FALSE
    ]
  }

  if (nrow(normalization_data) == 0) {
    return(return_without_norm(normalized))
  }

  normalization_row_ids <- normalization_data$.row_id

  for (col in available_columns) {
    numeric_values <- suppressWarnings(as.numeric(normalization_data[[col]]))
    participant_means <- ave(
      numeric_values,
      normalization_data$participant_id,
      FUN = function(x) {
        mean_value <- mean(x, na.rm = TRUE)
        if (is.nan(mean_value)) {
          return(NA_real_)
        }
        mean_value
      }
    )

    grand_mean <- mean(numeric_values, na.rm = TRUE)
    if (is.nan(grand_mean)) {
      grand_mean <- NA_real_
    }

    normalized_values <- numeric_values - participant_means + grand_mean
    normalized[[col]][normalization_row_ids] <- normalized_values
  }

  morey_factor <- if (expected_k > 1) sqrt(expected_k / (expected_k - 1)) else NA_real_
  normalized[[NORMALIZED_FLAG_COLUMN]][normalization_row_ids] <- TRUE
  normalized[[MOREY_FACTOR_COLUMN]][normalization_row_ids] <- morey_factor
  normalized[[K_CONDITIONS_COLUMN]][normalization_row_ids] <- expected_k

  return_without_norm(normalized)
}

ensure_likert_columns <- function(data, keys) {
  if (is.null(data) || nrow(data) == 0) {
    return(data)
  }

  working <- data
  parsed <- if ("response" %in% names(working)) {
    lapply(working$response, parse_response_json)
  } else {
    replicate(nrow(working), list(), simplify = FALSE)
  }

  for (key in keys) {
    existing <- if (key %in% names(working)) {
      suppressWarnings(as.numeric(working[[key]]))
    } else {
      rep(NA_real_, nrow(working))
    }

    if (!(key %in% names(working)) || sum(!is.na(existing)) == 0) {
      extracted <- suppressWarnings(as.numeric(
        vapply(parsed, function(item) safe_get_scalar(item[[key]]), character(1))
      ))

      valid <- extracted[!is.na(extracted)]
      if (length(valid) > 0 && all(valid >= 0 & valid <= 6)) {
        extracted <- extracted + 1
      }

      working[[key]] <- extracted
    } else {
      working[[key]] <- existing
    }
  }

  working
}

compute_composite_scores <- function(data) {
  if (is.null(data) || nrow(data) == 0) {
    return(data.frame())
  }

  working <- data
  for (col in c(TRUST_QUESTION_KEYS, INTERACTION_QUESTION_KEYS)) {
    if (col %in% names(working)) {
      working[[col]] <- suppressWarnings(as.numeric(working[[col]]))
    }
  }

  if (all(c(
    "skeptical_rating",
    "data_trust",
    "usability_difficulty",
    "comprehension_ease"
  ) %in% names(working))) {
    working$skeptical_rating_rev <- 8 - working$skeptical_rating
    working$usability_difficulty_rev <- 8 - working$usability_difficulty
    working$composite_trust_score <- rowMeans(
      working[, c(
        "skeptical_rating_rev",
        "data_trust",
        "usability_difficulty_rev",
        "comprehension_ease"
      ), drop = FALSE],
      na.rm = TRUE
    )
    working$composite_trust_score[is.nan(working$composite_trust_score)] <- NA_real_
  }

  interaction_available <- INTERACTION_QUESTION_KEYS[
    INTERACTION_QUESTION_KEYS %in% names(working)
  ]

  if (length(interaction_available) > 0) {
    working$composite_interaction_score <- rowMeans(
      working[, interaction_available, drop = FALSE],
      na.rm = TRUE
    )
    working$composite_interaction_score[is.nan(working$composite_interaction_score)] <- NA_real_
  }

  working
}

Fit Models and Compare (MSE, LOOCV-MSE, BIC, AIC, Lasso-Regularized MSE)

model_specs <- list(
  "Intercept-only (minimal)" = function(df) lm(trust ~ 1, data = df),
  "Linear" = function(df) lm(trust ~ lines, data = df),
  "Quadratic" = function(df) lm(trust ~ poly(lines, 2, raw = TRUE), data = df),
  "Cubic" = function(df) lm(trust ~ poly(lines, 3, raw = TRUE), data = df)
)

loocv_mse <- function(data, fit_fn) {
  n <- nrow(data)
  squared_errors <- rep(NA_real_, n)

  for (i in seq_len(n)) {
    train <- data[-i, , drop = FALSE]
    test <- data[i, , drop = FALSE]

    fit <- tryCatch(fit_fn(train), error = function(e) NULL)
    if (is.null(fit)) {
      next
    }

    prediction <- tryCatch(
      as.numeric(predict(fit, newdata = test)),
      error = function(e) NA_real_
    )

    squared_errors[i] <- (test$trust - prediction)^2
  }

  mean(squared_errors, na.rm = TRUE)
}

calc_approx_aic <- function(y, y_hat, k) {
  n <- length(y)
  rss <- sum((y - y_hat)^2)
  rss <- max(rss, .Machine$double.eps)
  n * log(rss / n) + n * (1 + log(2 * pi)) + 2 * k
}

calc_approx_bic <- function(y, y_hat, k) {
  n <- length(y)
  rss <- sum((y - y_hat)^2)
  rss <- max(rss, .Machine$double.eps)
  n * log(rss / n) + n * (1 + log(2 * pi)) + log(n) * k
}

get_bic_value <- function(model, y, y_hat) {
  bic_value <- tryCatch(as.numeric(BIC(model)), error = function(e) NA_real_)
  if (is.finite(bic_value)) {
    return(bic_value)
  }

  effective_k <- length(stats::coef(model)) + 1
  calc_approx_bic(y, y_hat, effective_k)
}

get_aic_value <- function(model, y, y_hat) {
  aic_value <- tryCatch(as.numeric(AIC(model)), error = function(e) NA_real_)
  if (is.finite(aic_value)) {
    return(aic_value)
  }

  effective_k <- length(stats::coef(model)) + 1
  calc_approx_aic(y, y_hat, effective_k)
}

calc_lasso_regularized_mse <- function(model, mse, lambda = LASSO_LAMBDA) {
  coefs <- tryCatch(stats::coef(model), error = function(e) numeric(0))
  if (length(coefs) == 0) {
    return(NA_real_)
  }

  coef_names <- names(coefs)
  if (is.null(coef_names)) {
    penalty <- sum(abs(coefs), na.rm = TRUE)
  } else {
    penalty <- sum(abs(coefs[coef_names != "(Intercept)"]), na.rm = TRUE)
  }

  mse + lambda * penalty
}

evaluate_model <- function(model_name, fit_fn, data) {
  fit <- fit_fn(data)
  fitted_values <- as.numeric(predict(fit, newdata = data))
  mse <- mean((data$trust - fitted_values)^2, na.rm = TRUE)
  loocv <- loocv_mse(data, fit_fn)
  bic_value <- get_bic_value(fit, data$trust, fitted_values)
  aic_value <- get_aic_value(fit, data$trust, fitted_values)
  lasso_reg_mse <- calc_lasso_regularized_mse(fit, mse)

  list(
    fit = fit,
    metrics = data.frame(
      model = model_name,
      mse = mse,
      loocv_mse = loocv,
      bic = bic_value,
      aic = aic_value,
      lasso_regularized_mse = lasso_reg_mse,
      stringsAsFactors = FALSE
    )
  )
}
